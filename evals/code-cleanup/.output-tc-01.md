_TAX_MULTIPLIERS: dict[int, float] = {
    1: 1.08,

}

def calculate_total(items: list[dict], discount: float | None = None) -> float:
    total = sum(
        item["val"] * _TAX_MULTIPLIERS.get(item["type"], 1.0)
        for item in items
    )
    if discount is not None:
        total -= discount
    return total
