### Suggested Fix

```typescript
import { Request, Response } from 'express';
import jwt from 'jsonwebtoken';

const JWT_SECRET = process.env.JWT_SECRET!;

export function authenticate(req: Request, res: Response): void {
  const authHeader = req.headers['authorization'];
  if (!authHeader) {
    res.status(401).send('Unauthorized');
    return;
  }

  const [, token] = authHeader.split(' ');
  try {
    const user = jwt.verify(token, JWT_SECRET);
    res.json({ success: true, user });
  } catch {
    res.status(401).send('Invalid token');
  }
}
```

Key changes: removes `eval` and the unsigned decode, uses `jsonwebtoken.verify()`, eliminates module-level mutable state, removes unused imports, adds return type annotation.
